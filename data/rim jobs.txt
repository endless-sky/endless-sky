mission "Southbound Shipment [0]"
	name "Shipment to <planet>"
	job
	repeat
	description "Deliver <cargo> to <destination>, the home of Southbound Shipyards, the largest ship manufacturer in the region. Payment is <payment>."
	cargo "weapons" 5 2 .1
	to offer
		random < 50
	source
		near "Zug" 2 10
		attributes "rim" "south"
	destination "Zug"
	npc
		government "Pirate"
		personality heroic harvests plunders
		fleet "Small Southern Pirates"
	on complete
		payment 0 160
		dialog "You drop off your cargo of <commodity> and collect your payment of <payment>."

mission "Han Sizer Month [0]"
	name `Han Sizer Celebration`
	job
	repeat
	description `Han Sizer month commemorates the first explorer of the Rim. To celebrate, citizens visit as many planets in the Rim as they can in the month. Bring <fare> to all the marked systems and return to <destination> for <payment>.`
	passengers 2 10 .9
	to offer
		random < 50
		month == 8
	source
		attributes "rim"
	stopover
		distance 2 4
		attributes "rim"

		distance 5 10
		attributes "rim"
