test-data "Npc Destination Mission"
	category mission
	contents
		mission "npc"
			landing
			destination "Geminus"
			npc land save
				ship "Quarg Wardragon"
				government "Quarg"
				destination Geminus
				waypoint
				personality waiting


test "Npc Destination Test"
	status active
	description "Starts the mission, accepts it and then goes to the destination making sure the Quarg got there first."
	sequence
		inject "Npc Destination Mission"
		inject "Npc Destination Save"
		call "Load First Savegame"
		# Set desired travel plan.
		navigate
			travel Castor
			travel Pollux
			travel Procyon
			travel Sirius
			"travel destination" Geminus
		call "Depart"
		# Trigger the mission.
		call "Land"
		call "Depart"
		input
			command jump
		label "moving"
		branch "moving"
			"flagship landed" != 1
		call "Depart"
		assert
			"npc: offered" == 1
			"npc: done" == 1
			"flagship system: Castor" == 1
