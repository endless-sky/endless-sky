	status active

	sequence

		# Set desired travel plan.
		navigate
			travel Betelgeuse
			travel Rigel
			travel Hassaleh

			travel "Ultima Thule"
			"travel destination" "Wormhole Alpha"
			travel Waypoint
			travel "Heia Due"
			travel Waypoint
			"travel destination" "Wormhole Alpha"
			travel "Ultima Thule"
			travel Rajak
			travel Hassaleh
			travel Rigel
